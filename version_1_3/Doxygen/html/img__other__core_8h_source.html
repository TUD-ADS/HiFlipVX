33 void ConvertDepthSingle(SrcType src, DstType &dst) {
34 #pragma HLS INLINE
35 
36  // Result
37  DstType result = 0;
38 
39  // Constants
40  const bool IN_IS_SIGNED = numeric_limits<SrcType>::is_signed;
41  const bool OUT_IS_SIGNED = numeric_limits<DstType>::is_signed;
42  const vx_int16 IN_SIZE = sizeof(SrcType);
43  const vx_int16 OUT_SIZE = sizeof(DstType);
44 
45  // Check for max shift value
46  const vx_uint16 MAX_SHIFT = (IN_SIZE > OUT_SIZE) ? static_cast<vx_uint16>(IN_SIZE - OUT_SIZE) : static_cast<vx_uint16>(OUT_SIZE - IN_SIZE);
47  STATIC_ASSERT(MAX_SHIFT*8 >= SHIFT, shift_size_out_of_range);
48 
49  // Check if down or up conversion
50  bool DOWN_CONVERSION = false;
51  bool UP_CONVERSION = false;
52 
53  // Border values for VX_CONVERT_POLICY_SATURATE
54  vx_int64 MAX_VAL = 0;
55  vx_int64 MIN_VAL = 0;
56 
57  // Set if down or up conversion
58  if ((IN_SIZE == 1 && OUT_SIZE == 2) ||
59  (IN_SIZE == 1 && OUT_SIZE == 4) ||
60  (IN_SIZE == 2 && OUT_SIZE == 4)) {
61  UP_CONVERSION = true;
62  } else if (
63  (IN_SIZE == 2 && OUT_SIZE == 1) ||
64  (IN_SIZE == 4 && OUT_SIZE == 1) ||
65  (IN_SIZE == 4 && OUT_SIZE == 2)) {
66  DOWN_CONVERSION = true;
67  }
68 
69  // Set border values for VX_CONVERT_POLICY_SATURATE
70  if (OUT_IS_SIGNED == true) {
71  if (OUT_SIZE == 1) {
72  MAX_VAL = VX_INT8_MAX;
73  MIN_VAL = VX_INT8_MIN;
74  } else if (OUT_SIZE == 2) {
75  MAX_VAL = VX_INT16_MAX;
76  MIN_VAL = VX_INT16_MIN;
77  } else if (OUT_SIZE == 4) {
78  MAX_VAL = VX_INT32_MAX;
79  MIN_VAL = VX_INT32_MIN;
80  }
81  } else if (OUT_IS_SIGNED == false) {
82  if (OUT_SIZE == 1)
83  MAX_VAL = VX_UINT8_MAX;
84  else if (OUT_SIZE == 2)
85  MAX_VAL = VX_UINT16_MAX;
86  else if (OUT_SIZE == 4)
87  MAX_VAL = VX_UINT32_MAX;
88  }
89 
90  // Do up or down conversion
91  if (DOWN_CONVERSION == true) {
92  if (CONV_POLICY == VX_CONVERT_POLICY_WRAP) {
93  result = (static_cast<DstType>(src >> SHIFT));
94  } else if (CONV_POLICY == VX_CONVERT_POLICY_SATURATE) {
95  vx_int64 value = static_cast<vx_int64>(src >> SHIFT);
96  value = max(min(value, MAX_VAL), MIN_VAL);
97  result = static_cast<DstType>(value);
98  }
99  }
100  if (UP_CONVERSION == true) {
101  if (CONV_POLICY == VX_CONVERT_POLICY_WRAP) {
102  result = (static_cast<DstType>(src)) << SHIFT;
103  } else if (CONV_POLICY == VX_CONVERT_POLICY_SATURATE) {
104  if (OUT_IS_SIGNED == true) {
105  vx_int64 value = (static_cast<vx_int64>(src)) << SHIFT;
106  value = static_cast<DstType>(max(min(value, MAX_VAL), MIN_VAL));
107  result = static_cast<DstType>(value);
108  } else {
109  result = (static_cast<DstType>(src)) << SHIFT;
110  }
111  }
112  }
113 
114  // Convert from signed to unsigned or v.v.
115  if ((IN_SIZE == OUT_SIZE) && (IN_IS_SIGNED != OUT_IS_SIGNED)) {
116  if (CONV_POLICY == VX_CONVERT_POLICY_WRAP) {
117  result = static_cast<DstType>(src);
118  } else if (CONV_POLICY == VX_CONVERT_POLICY_SATURATE) {
119  vx_int64 value = static_cast<vx_int64>(src);
120  value = max(min(value, MAX_VAL), MIN_VAL);
121  result = static_cast<DstType>(value);
122  }
123  }
124 
125  // Return result
126  dst = result;
127 }

867 template<vx_uint16 COLS_IN, vx_uint16 ROWS_IN, vx_uint16 COLS_OUT, vx_uint16 ROWS_OUT>
868 void ScaleDownNearest(vx_uint8 input[COLS_IN*ROWS_IN], vx_uint8 output[COLS_OUT*ROWS_OUT]) {
869 #pragma HLS INLINE
870 
871  // Compute the scale factor of the resized image in x and y direction
872  const vx_uint32 ACCURACY = 14;
873  const vx_uint32 COLS_SCALE = (vx_uint32)(((vx_float64)COLS_IN / (vx_float64)COLS_OUT) * pow(2, ACCURACY) + 0.5);
874  const vx_uint32 ROWS_SCALE = (vx_uint32)(((vx_float64)ROWS_IN / (vx_float64)ROWS_OUT) * pow(2, ACCURACY) + 0.5);
875 
876  // Compute the resized image
877  for (vx_uint32 y = 0, y_dst = 0, y_dst_next = 0; y < ROWS_IN; y++) {
878  for (vx_uint32 x = 0, x_dst = 0; x < COLS_IN; x++) {
879 #pragma HLS PIPELINE II=1
880 
881  // Get next input pixel in a stream
882  vx_uint8 data = input[y*COLS_IN + x];
883 
884  // Compute the coordinates of the next needed input pixel
885  vx_uint32 x_src = (((((x_dst << 1) + 1) * COLS_SCALE) >> ACCURACY) - 1) >> 1; // (vx_int32)floor((x_dst + 0.5f)*(COLS_IN/COLS_OUT) - 0.5f) using floating point computation
886  vx_uint32 y_src = (((((y_dst << 1) + 1) * ROWS_SCALE) >> ACCURACY) - 1) >> 1; // (vx_int32)floor((y_dst + 0.5f)*(ROWS_IN/ROWS_OUT) - 0.5f) using floating point computation
887 
888  // Check if current input pixel is the needed one
889  if ((y_src == y) && (x_src == x)) {
890 
891  // Write the ouput pixel to stream
892  output[y_dst*COLS_OUT + x_dst] = data;
893 
894  // Update pointer
895  x_dst++;
896  y_dst_next = 1;
897  }
898 
899  // Update pointer
900  if ((y_dst_next == 1) && (x == COLS_IN - 1)) {
901  y_dst++;
902  y_dst_next = 0;
903  }
904  }
905  }
906 }

916 template<vx_uint16 COLS_IN, vx_uint16 ROWS_IN, vx_uint16 COLS_OUT, vx_uint16 ROWS_OUT>
917 void ScaleDownBilinear(vx_uint8 input[COLS_IN*ROWS_IN], vx_uint8 output[COLS_OUT*ROWS_OUT]) {
918 #pragma HLS INLINE
919 
920  // Compute constants
921  const vx_uint32 ACCURACY = 12; // Minimum 12, Maximum 15
922  const vx_uint32 ONE = (1 << ACCURACY);
923  const vx_uint32 SHIFT = 2*ACCURACY - 10-8;
924  const vx_uint32 MASK_FRAKTION = (1 << (ACCURACY + 2)) - 1;
925  const vx_uint32 COLS_SCALE = (vx_uint32)(((vx_float64)COLS_IN / (vx_float64)COLS_OUT) * pow(2, ACCURACY) + 0.5);
926  const vx_uint32 ROWS_SCALE = (vx_uint32)(((vx_float64)ROWS_IN / (vx_float64)ROWS_OUT) * pow(2, ACCURACY) + 0.5);
927 
928  // Strt and step size of src ptr with "ACCURACY" fraction size
929  // x_src = (x + 0.5) * scale_cols - 0.5 | fraction = 1+1+ACCURACY | not negative: SCALE > 1.0f
930  const vx_uint32 X_SRC_STRT = (COLS_SCALE << 1) - 1;
931  const vx_uint32 Y_SRC_STRT = (ROWS_SCALE << 1) - 1;
932  const vx_uint32 X_SRC_STEP = COLS_SCALE * 4;
933  const vx_uint32 Y_SRC_STEP = ROWS_SCALE * 4;
934 
935  // Buffer the input pixel
936  vx_uint8 data_new = 0;
937  vx_uint8 data_left = 0, data_right = 0;
938  vx_uint8 buffer_one[COLS_IN];
939  vx_uint8 buffer_two[COLS_IN];
940 
941  vx_uint32 in_cnt = 0, out_cnt = 0;
942 
943  // Compute the resized image
944  for (vx_uint32 y = 0, y_dst = 0, y_src = Y_SRC_STRT, y_dst_next = 0; y <= ROWS_IN; y++) {
945  for (vx_uint32 x = 0, x_dst = 0, x_src = X_SRC_STRT; x <= COLS_IN; x++) {
946 #pragma HLS PIPELINE II=1
947 
948  vx_uint8 tl, tr, bl, br;
949 
950  // Buffer the previous input pixel [x-1]
951  vx_uint8 data_old = data_new;
952 
953  // Read Input Pixel from Stream
954  if (x < COLS_IN && y < ROWS_IN) {
955  in_cnt++;
956  data_new = input[y*COLS_IN + x];
957  }
958 
959  // Compute the coordinates of all needed input pixels
960  vx_uint32 x_fract = (x_src & MASK_FRAKTION) >> 2;
961  vx_uint32 y_fract = (y_src & MASK_FRAKTION) >> 2;
962  vx_uint16 x_l = (vx_uint16)(x_src >> (ACCURACY + 2));
963  vx_uint16 y_t = (vx_uint16)(y_src >> (ACCURACY + 2));
964  vx_uint16 x_r = x_l + 1;
965  vx_uint16 y_b = y_t + 1;
966 
967 
968 
969  // Write/Read data to/from buffer
970  if ((y % 2) == 0) {
971  if (x_l < COLS_IN)
972  data_left = buffer_two[x_l];
973  if (x_r < COLS_IN)
974  data_right = buffer_two[x_r];
975  if (x > 0)
976  buffer_one[x - 1] = data_old;
977  } else {
978  if (x_l < COLS_IN)
979  data_left = buffer_one[x_l];
980  if (x_r < COLS_IN)
981  data_right = buffer_one[x_r];
982  if (x > 0)
983  buffer_two[x - 1] = data_old;
984  }
985 
986  // Get the 4 needed input pixel (using replicated borders)
987  if (y_t == COLS_IN - 1) {
988  if (x_l == ROWS_IN - 1) {
989  tl = tr = bl = br = data_left;
990  } else {
991  tr = br = data_right;
992  tl = bl = data_left;
993  }
994  } else {
995  if (x_l == ROWS_IN - 1) {
996  bl = br = data_old;
997  tl = tr = data_left;
998  } else {
999  tl = data_right;
1000  tr = data_left;
1001  bl = data_old;
1002  br = data_new;
1003  }
1004  }
1005 
1006  // Compute the Bilinear Interpolation (Saturated)
1007  vx_uint16 tl_part = (vx_uint16)(((((ONE - x_fract) * (ONE - y_fract)) >> 10) * (vx_uint32)tl) >> 8);
1008  vx_uint16 tr_part = (vx_uint16)((((( x_fract) * (ONE - y_fract)) >> 10) * (vx_uint32)tr) >> 8);
1009  vx_uint16 bl_part = (vx_uint16)(((((ONE - x_fract) * ( y_fract)) >> 10) * (vx_uint32)bl) >> 8);
1010  vx_uint16 br_part = (vx_uint16)((((( x_fract) * ( y_fract)) >> 10) * (vx_uint32)br) >> 8);
1011  vx_uint16 sum = (tl_part + tr_part + bl_part + br_part) >> (2 * ACCURACY - 10 - 8);
1012  vx_uint8 result = (vx_uint8)(min(sum, (vx_uint16)(255)));
1013 
1014  // Check if the input pixel are the needed ones
1015  if ((x_r == x) && (y_b == y)) {
1016  out_cnt++;
1017 
1018  // Write ouput pixel to stream
1019  output[y_dst*COLS_OUT + x_dst] = result;
1020 
1021  // Update pointer
1022  x_dst++;
1023  x_src += X_SRC_STEP;
1024  y_dst_next = 1;
1025  }
1026 
1027  // Update pointer
1028  if ((y_dst_next == 1) && (x_dst == COLS_OUT)) {
1029  y_dst++;
1030  y_src += Y_SRC_STEP;
1031  y_dst_next = 0;
1032  }
1033  }
1034  }
1035 }

1124 template<typename SrcType, const vx_uint32 IMG_PIXELS, const vx_uint32 DISTRIBUTION_BINS, const vx_uint32 DISTRIBUTION_RANGE, const SrcType DISTRIBUTION_OFFSET>
1125 void Histogram(SrcType input[IMG_PIXELS], vx_uint32 output[IMG_PIXELS]) {
1126 #pragma HLS INLINE
1127 
1128  // Check Datatype, MaximumValue, and that BIN is smaller RANGE
1129  const vx_type_e TYPE = GET_TYPE(SrcType);
1130  const bool allowed_type = (TYPE == VX_TYPE_UINT8) || (TYPE == VX_TYPE_UINT16);
1131  const vx_uint32 MAX_VAL = (TYPE == VX_TYPE_UINT8) ? (static_cast<vx_uint32>(0x100)) : (static_cast<vx_uint32>(0x10000));
1132  const bool allowed_value1 = ((DISTRIBUTION_BINS <= MAX_VAL) && (DISTRIBUTION_RANGE <= MAX_VAL));
1133  const bool allowed_value2 = (static_cast<vx_uint32>(DISTRIBUTION_OFFSET) <= (MAX_VAL - static_cast<vx_uint32>(1)));
1134  const bool allowed_range = DISTRIBUTION_BINS <= DISTRIBUTION_RANGE;
1135  STATIC_ASSERT(allowed_type, data_type_is_not_allowed_for_histogram);
1136  STATIC_ASSERT(allowed_value1, BINS_RANGE_value_out_of_range_max_0x100_for_uint8_and_0x10000_for_uint16);
1137  STATIC_ASSERT(allowed_value2, OFFSET_value_out_of_range_max_0xff_for_uint8_and_0xffff_for_uint16);
1138  STATIC_ASSERT(allowed_range, RANGE_should_be_smaller_or_equal_to_BINS);
1139 
1140  // Compute Scale (BINS / RANGE) "approximation"
1141  const vx_uint16 ACCURACY = 15;
1142  const vx_uint16 SCALE = static_cast<vx_uint16>((DISTRIBUTION_BINS << ACCURACY) / DISTRIBUTION_RANGE);
1143 
1144  // Global variables
1145  vx_uint16 pre_bin = 0;
1146  vx_uint32 pre_val = 0;
1147 
1148  // Buffer Histogram in BRAM
1149  vx_uint32 LUTA[DISTRIBUTION_BINS];
1150  vx_uint32 LUTB[DISTRIBUTION_BINS];
1151 
1152  // Reset histogram
1153  for (vx_uint32 i = 0; i < DISTRIBUTION_BINS; i++) {
1154 #pragma HLS PIPELINE II=1
1155  LUTA[i] = 0;
1156  LUTB[i] = 0;
1157  }
1158 
1159  // Compute Histogram
1160  for (vx_uint32 i = 0; i < IMG_PIXELS + 1; i++) {
1161 #pragma HLS PIPELINE II=1
1162 
1163  // Variables
1164  vx_uint16 cur_bin = 0;
1165  vx_uint32 cur_val = 0;
1166  SrcType pixel = 0;
1167  vx_bool range = vx_false_e;
1168 
1169  // Read input data
1170  if (i < IMG_PIXELS)
1171  pixel = input[i];
1172 
1173  // Check if bin will be out of range
1174  if ((DISTRIBUTION_OFFSET <= pixel) && (pixel <= (DISTRIBUTION_OFFSET + DISTRIBUTION_RANGE)) && (i < IMG_PIXELS))
1175  range = vx_true_e;
1176 
1177  // Subtract offset and compute current bin bin
1178  if (range) {
1179  const SrcType index = pixel - DISTRIBUTION_OFFSET;
1180 
1181  if (DISTRIBUTION_BINS != DISTRIBUTION_RANGE)
1182  cur_bin = static_cast<vx_uint16>((static_cast<vx_uint32>(index) * static_cast<vx_uint32>(SCALE)) >> ACCURACY);
1183  else
1184  cur_bin = static_cast<vx_uint16>(index);
1185  }
1186 
1187  // Read current histogram entry & write previous histogram entry
1188  if (i % 2 == 0) {
1189  cur_val = LUTA[cur_bin];
1190  LUTB[pre_bin] = pre_val;
1191  } else {
1192  cur_val = LUTB[cur_bin];
1193  LUTA[pre_bin] = pre_val;
1194  }
1195 
1196  // Increment histogram value
1197  if (range)
1198  cur_val++;
1199 
1200  // Update to store in next iteration
1201  pre_bin = cur_bin;
1202  pre_val = cur_val;
1203  }
1204 
1205  // Write output data
1206  for (vx_uint32 i = 0; i < DISTRIBUTION_BINS; i++) {
1207 #pragma HLS PIPELINE II = 1
1208  output[i] = LUTA[i] + LUTB[i];
1209  }
1210 }
