299 template<typename OutType, typename InType, bool CHECK_MAX, vx_round_policy_e ROUND_POLICY>
300 OutType SqrtLester(InType value) {
301 #pragma HLS INLINE
302 
303  const OutType MAX_VAL = ComputeMax<OutType>();
304 
305  // Number of stages (latency)
306  const vx_uint8 N = sizeof(OutType) * 8;
307 
308  // Variables
309  OutType A1 = 0; // A^1 Intermediate result
310  InType A2 = 0; // A^2 Square of the intermediate result
311 
312  // Each stage computes 1 bit of the resulting vector
313  for (vx_uint8 n = N - 1; n < N; n--) {
314 #pragma HLS unroll
315 
316  // Add new bit of position n and compute (A1 + B1)^2
317  OutType B1 = static_cast<OutType>( 1) << (n);
318  InType B2 = static_cast<InType >(B1) << (n);
319  InType AB = static_cast<InType >(A1) << (n);
320  InType A2_next = A2 + B2 + (AB << 1); // A*A + B*B + 2*A*B
321 
322  // Store if tmp does not exceed value
323  if (A2_next <= value) {
324  A1 |= B1;
325  A2 = A2_next;
326  }
327  }
328 
329  // Round to the nearest integer and check for overflow
330  if (ROUND_POLICY == VX_ROUND_POLICY_TO_NEAREST_EVEN) {
331  if (CHECK_MAX) {
332  if (((value - A2) > static_cast<InType>(A1)) && (A1 != MAX_VAL))
333  A1++;
334  } else {
335  if ((value - A2) > static_cast<InType>(A1))
336  A1++;
337  }
338  }
339 
340  // Return result
341  return A1;
342 }

355 template <typename ScalarType, const vx_uint8 KERN_SIZE>
356 const vx_uint16 ComputeGaussianKernelSigma(ScalarType kernel[KERN_SIZE][KERN_SIZE], const double sigma) {
357 #pragma HLS INLINE
358 
359  // Constants
360  const vx_uint16 K_RAD = (vx_uint16)KERN_SIZE / (vx_uint16)2;
361  const double s = 2.0 * sigma * sigma;
362 
363  // Variables
364  double sum = 0.0;
365  double doubleKernel[KERN_SIZE][KERN_SIZE];
366 #pragma HLS array_partition variable=doubleKernel complete dim=0
367 
368  // Creates the gaussian kernel
369  for (vx_uint8 x = 0; x < KERN_SIZE; x++) {
370 #pragma HLS unroll
371  for (vx_uint8 y = 0; y < KERN_SIZE; y++) {
372 #pragma HLS unroll
373  const vx_uint32 a = static_cast<vx_uint32>((vx_uint16)x - K_RAD);
374  const vx_uint32 b = static_cast<vx_uint32>((vx_uint16)y - K_RAD);
375  const double c = static_cast<double>(a*a + b*b);
376  const double r = sqrt(c);
377  doubleKernel[x][y] = (exp(-(r*r) / s)) / (M_PI * s);
378  sum += doubleKernel[x][y];
379  }
380  }
381 
382  // Normalizes the gaussian kernel
383  for (vx_uint8 x = 0; x < KERN_SIZE; x++) {
384 #pragma HLS unroll
385  for (vx_uint8 y = 0; y < KERN_SIZE; y++) {
386 #pragma HLS unroll
387  doubleKernel[x][y] /= sum;
388  }
389  }
390 
391  // Computes the fraction for the fixed point representation
392  vx_int32 e_max = VX_INT32_MIN;
393  for (vx_uint8 i = 0; i < KERN_SIZE; i++) {
394 #pragma HLS unroll
395  for (vx_uint8 j = 0; j < KERN_SIZE; j++) {
396 #pragma HLS unroll
397  int e = 0;
398  frexp(doubleKernel[i][j], &e);
399  e_max = std::max(e_max, (vx_int32)e);
400  }
401  }
402  const vx_int32 fraction = 8 * sizeof(ScalarType) - e_max;
403  const vx_uint16 kernFraction = static_cast<vx_uint16>(std::min(std::max(fraction, (vx_int32)0), (vx_int32)31));
404 
405  // Computes and stores the fixed point kernel
406  vx_uint32 shift = 1;
407  for (vx_uint16 i = 0; i < kernFraction; i++) {
408 #pragma HLS unroll
409  shift *= 2;
410  }
411  for (vx_uint8 i = 0; i < KERN_SIZE; i++) {
412 #pragma HLS unroll
413  for (vx_uint8 j = 0; j < KERN_SIZE; j++) {
414 #pragma HLS unroll
415  kernel[i][j] = static_cast<ScalarType>(doubleKernel[i][j] * static_cast<double>(shift));
416  }
417  }
418 
419  // Returns the fraction
420  return kernFraction;

554 void ComputeNormalization(const vx_uint64 kernel_sum, vx_uint64 &norm_mult, vx_uint32 &norm_shift) {
555 #pragma HLS INLINE
556 
557  // Constants
558  const vx_uint32 MAX_SHIFT = static_cast<vx_uint32>(48);
559  const vx_uint32 MIN_SHIFT = static_cast<vx_uint32>(15);
560  const vx_uint64 ONE = static_cast<vx_uint64>(1);
561  const vx_uint64 ZERO = static_cast<vx_uint64>(0);
562 
563  // Variables
564  vx_uint16 new_shift = 0;
565  vx_uint16 new_mult = 1;
566  vx_uint32 old_shift = 0;
567  vx_uint64 old_mult = 1;
568  vx_uint64 norm_mult_a = 1;
569  vx_uint32 norm_shift_a = 0;
570  vx_uint64 norm_mult_b = 1;
571  vx_uint32 norm_shift_b = 0;
572  vx_uint64 temp = 0;
573  bool is_power_two = false;
574 
575  // Compute Mult and Shift (Not Power of two)
576  old_mult = temp = (ONE << MAX_SHIFT) / kernel_sum;
577  for (vx_uint32 i = 0; i < MAX_SHIFT; i++) {
578 #pragma HLS unroll
579  if (temp > ZERO) {
580  temp = temp >> static_cast<vx_uint32>(1);
581  old_shift = i;
582  }
583  }
584  if (old_shift > 16) {
585  new_shift = static_cast<vx_uint16>(MAX_SHIFT - (old_shift - MIN_SHIFT));
586  new_mult = static_cast<vx_uint16>(old_mult >> (old_shift - MIN_SHIFT));
587  } else {
588  new_shift = static_cast<vx_uint16>(MAX_SHIFT);
589  new_mult = static_cast<vx_uint16>(old_mult);
590  }
591  norm_mult_a = static_cast<vx_uint64>(new_mult);
592  norm_shift_a = static_cast<vx_uint32>(new_shift);
593 
594  // Compute Mult and Shift (Power of two)
595  for (vx_uint32 i = 0; i < sizeof(vx_uint32) * 8; i++) {
596 #pragma HLS unroll
597  if ((kernel_sum & (ONE << i)) != ZERO)
598  norm_shift_b = i;
599  }
600 
601  // Check if scaling is by a power of two
602  is_power_two = (kernel_sum & (kernel_sum - ONE)) == ZERO;
603 
604  // Compute Normalization and Multiplication
605  norm_mult = (is_power_two) ? (norm_mult_b) : (norm_mult_a);
606  norm_shift = (is_power_two) ? (norm_shift_b) : (norm_shift_a);
607 }
