217 template<typename Type, const vx_uint16 WIDTH, const vx_uint16 HEIGHT, const vx_uint16 KERN_SIZE, const vx_border_e BORDER_TYPE>
218 void SwBox(Type * input, Type * output) {
219 
220  // Constants
221  const vx_int32 KERN_RAD = KERN_SIZE / 2;
222  const float NORM = 1.0f / powf(KERN_SIZE, 2);
223 
224  // Compute box filter
225  for (vx_int32 y = 0; y < HEIGHT; y++) {
226  for (vx_int32 x = 0; x < WIDTH; x++) {
227 
228  // Data is passed through if filter exceeds border
229  if (BORDER_TYPE == VX_BORDER_UNDEFINED) {
230  if (x < KERN_RAD || x >= WIDTH - KERN_RAD || y < KERN_RAD || y >= HEIGHT - KERN_RAD) {
231  output[y*WIDTH + x] = input[y*WIDTH + x];
232  continue;
233  }
234  }
235 
236  // Compute single box filter
237  float sum = 0.0f;
238  for (vx_int32 i = -KERN_RAD; i <= KERN_RAD; i++) {
239  for (vx_int32 j = -KERN_RAD; j <= KERN_RAD; j++) {
240 
241  // Replicated border
242  if (BORDER_TYPE == VX_BORDER_REPLICATE) {
243  vx_int32 yi = min(max(y + i, static_cast<vx_int32>(0)), (vx_int32)(HEIGHT - 1));
244  vx_int32 xj = min(max(x + j, static_cast<vx_int32>(0)), (vx_int32)(WIDTH - 1));
245  sum += static_cast<float>(input[yi*WIDTH + xj]);
246 
247  // Constant border
248  } else if (BORDER_TYPE == VX_BORDER_CONSTANT) {
249  vx_int32 yi = y + i;
250  vx_int32 xj = x + j;
251  if ((xj >= 0) && (xj < WIDTH) && (yi >= 0) && (yi < HEIGHT))
252  sum += static_cast<float>(input[yi*WIDTH + xj]);
253 
254  // Data is passed through if filter exceeds border
255  } else if (BORDER_TYPE == VX_BORDER_UNDEFINED) {
256  vx_int32 yi = y + i;
257  vx_int32 xj = x + j;
258  sum += static_cast<float>(input[yi*WIDTH + xj]);
259  }
260  }
261  }
262 
263  // Store result
264  output[y*WIDTH + x] = static_cast<Type>(sum * NORM + 0.5f);
265  }
266  }
